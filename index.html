const USERNAME = 'valdrea'; const REPO = 'My-Games'; // Quick helper function to convert bytes to MB const formatSize = (bytes) => { return (bytes / (1024 * 1024)).toFixed(1) + ' MB'; }; async function scanGames() { const container = document.getElementById('game-container'); const status = document.getElementById('status'); try { const response = await fetch(`https://api.github.com/repos/${USERNAME}/${REPO}/contents/`); const items = await response.json(); status.style.display = 'none'; const folders = items.filter(item => item.type === 'dir' && !item.name.startsWith('.')); for (const folder of folders) { const contentResponse = await fetch(folder.url); const contents = await contentResponse.json(); const hasWeb = contents.some(file => file.name.toLowerCase() === 'index.html'); const exeFiles = contents.filter(file => file.name.toLowerCase().endsWith('.exe')); const audioFile = exeFiles.find(f => f.name.toLowerCase() === 'audiopack.exe'); const gameExe = exeFiles.find(f => f.name.toLowerCase() !== 'audiopack.exe'); const card = document.createElement('div'); card.className = 'game-card'; let buttonHTML = ''; if (hasWeb) { buttonHTML = `[ PLAY WEB VERSION ]`; } else { // Add size to Game/Installer button if (gameExe) { const btnLabel = gameExe.name.toLowerCase().includes('install') ? 'INSTALLER' : 'GAME .EXE'; const sizeText = formatSize(gameExe.size); buttonHTML += `[ DOWNLOAD ${btnLabel} - ${sizeText} ]`; } // Add size to Audio Pack button if (audioFile) { const sizeText = formatSize(audioFile.size); buttonHTML += `[ DOWNLOAD AUDIO PACK - ${sizeText} ]`; } } if (!buttonHTML) buttonHTML = `
[ DATA SCAN COMPLETE: NO BOOTABLE MEDIA ]
`; card.innerHTML = `
${folder.name.replace(/-/g, ' ')}
${buttonHTML}
`; container.appendChild(card); startSlideshow(`slideshow-${folder.name}`); } } catch (error) { status.innerHTML = "FATAL ERROR: NEURAL LINK DISCONNECTED."; } } function startSlideshow(containerId) { setTimeout(() => { const images = Array.from(document.querySelectorAll(`#${containerId} img`)).filter(img => img.style.display !== 'none'); if (images.length <= 1) return; let currentIndex = 0; setInterval(() => { images[currentIndex].classList.remove('active'); currentIndex = (currentIndex + 1) % images.length; images[currentIndex].classList.add('active'); }, 3500); }, 1000); } scanGames();
